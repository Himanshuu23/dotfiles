bash#!/bin/bash

# Check if Spotify is running at all
if ! pgrep -x spotify > /dev/null 2>&1; then
    echo '{"text": "", "class": "hidden"}'
    exit 0
fi

# Get player status
status=$(playerctl --player=spotify status 2>/dev/null)

# If no status (Spotify open but no song loaded)
if [ -z "$status" ] || [ "$status" = "Stopped" ]; then
    echo '{"text": "<span size=\"13000\" foreground=\"#1DB954\"></span>", "tooltip": "Spotify", "class": "idle"}'
    exit 0
fi

# Get song info for tooltip
artist=$(playerctl --player=spotify metadata artist 2>/dev/null | head -c 50)
title=$(playerctl --player=spotify metadata title 2>/dev/null | head -c 50)

# If playing, show rotating disk
if [ "$status" = "Playing" ]; then
    echo "{\"text\": \"<span size='15000' foreground='#1DB954'>Û∞ùö</span>\", \"tooltip\": \"$title - $artist\", \"class\": \"playing\"}"
# If paused, show static disk
elif [ "$status" = "Paused" ]; then
    echo "{\"text\": \"<span size='15000' foreground='#888888'>Û∞ùö</span>\", \"tooltip\": \"$title - $artist (Paused)\", \"class\": \"idle\"}"
else
    # Fallback: show normal Spotify icon
    echo '{"text": "<span size=\"13000\" foreground=\"#1DB954\"></span>", "tooltip": "Spotify", "class": "idle"}'
fi

2. spotify_tooltip.sh
bash#!/bin/bash

# Get current song info
status=$(playerctl --player=spotify status 2>/dev/null)
artist=$(playerctl --player=spotify metadata artist 2>/dev/null)
title=$(playerctl --player=spotify metadata title 2>/dev/null)
album=$(playerctl --player=spotify metadata album 2>/dev/null)
art_url=$(playerctl --player=spotify metadata mpris:artUrl 2>/dev/null)

# Download album art
art_file="/tmp/spotify_tooltip_art.jpg"
if [ -n "$art_url" ]; then
    curl -s "$art_url" -o "$art_file" 2>/dev/null
fi

# Status icon
if [ "$status" = "Playing" ]; then
    status_icon="‚ñ∂"
else
    status_icon="‚è∏"
fi

# Create tooltip using yad
yad --image="$art_file" \
    --title="Spotify" \
    --text="<big><b>$title</b></big>\n<i>$artist</i>\n<small>$album</small>\n\n<b>$status_icon $status</b>" \
    --button="‚èÆ:1" \
    --button="‚èØ:2" \
    --button="‚è≠:3" \
    --width=350 \
    --height=180 \
    --borders=20 \
    --center \
    --undecorated \
    --skip-taskbar \
    --on-top \
    --timeout=10 \
    --timeout-indicator=bottom

BUTTON=$?

case $BUTTON in
    1) playerctl --player=spotify previous ;;
    2) playerctl --player=spotify play-pause ;;
    3) playerctl --player=spotify next ;;
esac

bash#!/bin/bash

# Function to get icon path from .desktop file
get_icon_path() {
    local app_name=$1
    local desktop_file=""
    
    # Search for .desktop file
    for dir in /usr/share/applications ~/.local/share/applications /var/lib/flatpak/exports/share/applications; do
        if [ -d "$dir" ]; then
            desktop_file=$(find "$dir" -iname "*${app_name}*.desktop" 2>/dev/null | head -n 1)
            [ -n "$desktop_file" ] && break
        fi
    done
    
    if [ -z "$desktop_file" ]; then
        echo ""
        return
    fi
    
    # Extract icon name
    icon_name=$(grep -m 1 "^Icon=" "$desktop_file" | cut -d'=' -f2)
    
    # If icon is full path, return it
    if [ -f "$icon_name" ]; then
        echo "$icon_name"
        return
    fi
    
    # Search for icon in icon themes
    for size in scalable 48x48 64x64 128x128 256x256; do
        for theme_dir in /usr/share/icons/hicolor /usr/share/icons/Papirus /usr/share/pixmaps ~/.local/share/icons; do
            if [ -d "$theme_dir" ]; then
                icon_file=$(find "$theme_dir" -name "${icon_name}.*" -path "*/${size}/*" 2>/dev/null | head -n 1)
                if [ -n "$icon_file" ]; then
                    echo "$icon_file"
                    return
                fi
            fi
        done
    done
    
    # Last resort: check pixmaps
    if [ -f "/usr/share/pixmaps/${icon_name}.png" ]; then
        echo "/usr/share/pixmaps/${icon_name}.png"
    elif [ -f "/usr/share/pixmaps/${icon_name}.svg" ]; then
        echo "/usr/share/pixmaps/${icon_name}.svg"
    else
        echo ""
    fi
}

# Check if app is running
is_running() {
    pgrep -i "$1" > /dev/null 2>&1
}

# Define dock apps (process_name:search_name:fallback_icon:command)
apps=(
    "ghostty:ghostty::ghostty"
    "zen:zen-browser:Û∞àπ:zen-browser"
    "systemsettings:systemsettings::systemsettings5"
    "spotify:spotify::spotify"
)

# Build dock HTML
output=""

for app in "${apps[@]}"; do
    IFS=':' read -r process search_name fallback_icon command <<< "$app"
    
    # Get icon
    icon_path=$(get_icon_path "$search_name")
    
    # Build icon HTML
    if [ -n "$icon_path" ] && [ -f "$icon_path" ]; then
        icon_html="<img src='file://$icon_path'/>"
    elif [ -n "$fallback_icon" ]; then
        icon_html="<span>$fallback_icon</span>"
    else
        icon_html="<span></span>"
    fi
    
    # Check running status
    if is_running "$process"; then
        output+="<span class='dock-item running' onclick='hyprctl dispatch focuswindow $process'>$icon_html</span>"
    else
        output+="<span class='dock-item' onclick='$command &'>$icon_html</span>"
    fi
done
echo "{\"text\": \"$output\", \"tooltip\": \"Application Dock\", \"class\": \"dock\"}"
